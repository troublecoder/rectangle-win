```rust
use std::path::{Path, PathBuf};
use std::sync::Mutex;

use crate::application::errors::{ApplicationError, AppResult};
use crate::application::ports::ConfigStore;
use crate::domain::model::Config;

/// TOML кё°л°ҳ м„Өм • м ҖмһҘмҶҢ.
/// кІҪлЎң: %APPDATA%/rectangle-win/config.toml (лҳҗлҠ” н”Ңлһ«нҸјлі„ config_dir)
pub struct TomlConfigStore {
    path: PathBuf,
    /// нҢҢмқј мқҪкё° м „ мәҗмӢң (ліҖкІҪмӢң save()лЎң к°ұмӢ )
    cache: Mutex<Option<Config>>,
}

impl TomlConfigStore {
    /// кё°ліё кІҪлЎңлЎң мғқм„ұ: {config_dir}/rectangle-win/config.toml
    pub fn default_path() -> Self {
        let config_dir = dirs::config_dir()
            .unwrap_or_else(|| PathBuf::from("."));
        let dir = config_dir.join("rectangle-win");
        // л””л үнҶ лҰ¬к°Җ м—Ҷмңјл©ҙ мғқм„ұ
        let _ = std::fs::create_dir_all(&dir);
        Self {
            path: dir.join("config.toml"),
            cache: Mutex::new(None),
        }
    }

    /// лӘ…мӢңм Ғ кІҪлЎңлЎң мғқм„ұ (н…ҢмҠӨнҠёмҡ©)
    pub fn at_path(path: impl AsRef<Path>) -> Self {
        Self {
            path: path.as_ref().to_path_buf(),
            cache: Mutex::new(None),
        }
    }

    fn read_from_disk(&self) -> AppResult<Config> {
        if !self.path.exists() {
            // нҢҢмқјмқҙ м—Ҷмңјл©ҙ кё°ліёк°’мңјлЎң мғқм„ұ
            let default = Config::default();
            self.write_to_disk(&default)?;
            return Ok(default);
        }
        let contents = std::fs::read_to_string(&self.path)
            .map_err(|e| ApplicationError::WindowOperation(
                format!("failed to read config: {}", e)
            ))?;
        let config: Config = toml::from_str(&contents)
            .map_err(|e| ApplicationError::WindowOperation(
                format!("failed to parse config: {}", e)
            ))?;
        Ok(config)
    }

    fn write_to_disk(&self, config: &Config) -> AppResult<()> {
        let contents = toml::to_string_pretty(config)
            .map_err(|e| ApplicationError::WindowOperation(
                format!("failed to serialize config: {}", e)
            ))?;
        std::fs::write(&self.path, contents)
            .map_err(|e| ApplicationError::WindowOperation(
                format!("failed to write config: {}", e)
            ))?;
        Ok(())
    }
}

impl ConfigStore for TomlConfigStore {
    fn load(&self) -> AppResult<Config> {
        let mut cache = self.cache.lock().unwrap();
        if let Some(ref cached) = *cache {
            return Ok(cached.clone());
        }
        let config = self.read_from_disk()?;
        *cache = Some(config.clone());
        Ok(config)
    }

    fn save(&self, config: &Config) -> AppResult<()> {
        self.write_to_disk(config)?;
        *self.cache.lock().unwrap() = Some(config.clone());
        Ok(())
    }

    fn path(&self) -> &Path {
        &self.path
    }
}
```

```rust
#![cfg(windows)]

use windows::Win32::Foundation::{HWND, RECT};
use windows::Win32::Graphics::Gdi::{MonitorFromWindow, MONITOR_DEFAULTTONEAREST, GetMonitorInfoW, MONITORINFOEXW};
use windows::Win32::UI::WindowsAndMessaging::{
    GetForegroundWindow, GetWindowRect, SetWindowPos, MoveWindow,
    ShowWindow, SW_MINIMIZE, SW_MAXIMIZE, SW_RESTORE,
    HWND_TOP, SWP_NOZORDER, SWP_SHOWWINDOW,
};

use crate::application::errors::{ApplicationError, AppResult};
use crate::application::ports::WindowMover;
use crate::domain::geometry::{MonitorBounds, Pixel};
use crate::domain::model::{SnapTarget, WindowAction};
use euclid::Rect;

pub struct Win32WindowMover;

impl Win32WindowMover {
    pub fn new() -> Self { Self }

    fn hwnd_from_u64(&self, handle: u64) -> HWND {
        HWND(handle as *mut _)
    }

    fn rect_to_monitor(&self, rect: RECT) -> MonitorBounds {
        MonitorBounds::new(
            rect.left, rect.top,
            rect.right - rect.left,
            rect.bottom - rect.top,
        )
    }
}

impl WindowMover for Win32WindowMover {
    fn get_foreground_window(&self) -> Option<u64> {
        let hwnd = unsafe { GetForegroundWindow() };
        if hwnd.0.is_null() { None } else { Some(hwnd.0 as u64) }
    }

    fn apply_snap_target(
        &self,
        window_handle: u64,
        target: &SnapTarget,
        monitor: &MonitorBounds,
    ) -> AppResult<()> {
        let hwnd = self.hwnd_from_u64(window_handle);

        match target {
            SnapTarget::Area { x_ratio, y_ratio, w_ratio, h_ratio, .. } => {
                let pixel_rect = crate::domain::geometry::ratio_to_pixels(
                    *x_ratio, *y_ratio, *w_ratio, *h_ratio, monitor,
                );
                // SetWindowPosлЎң мқҙлҸҷ + нҒ¬кё° лҸҷмӢң м„Өм •
                unsafe {
                    SetWindowPos(
                        hwnd, HWND_TOP,
                        pixel_rect.origin.x, pixel_rect.origin.y,
                        pixel_rect.size.width, pixel_rect.size.height,
                        SWP_NOZORDER | SWP_SHOWWINDOW,
                    ).map_err(|e| ApplicationError::WindowOperation(
                        format!("SetWindowPos failed: {}", e)
                    ))?;
                }
            }
            SnapTarget::Action { action, .. } => {
                // м•Ўм…ҳ мӢӨн–ү
                match action {
                    WindowAction::Maximize => unsafe {
                        let _ = ShowWindow(hwnd, SW_MAXIMIZE);
                    },
                    WindowAction::Minimize => unsafe {
                        let _ = ShowWindow(hwnd, SW_MINIMIZE);
                    },
                    WindowAction::Restore => unsafe {
                        let _ = ShowWindow(hwnd, SW_RESTORE);
                    },
                    WindowAction::Center => {
                        // нҳ„мһ¬ м°Ҫ нҒ¬кё° мң м§Җ + мӨ‘м•ҷ мқҙлҸҷ
                        if let Ok(rect) = self.get_window_rect(window_handle) {
                            let cx = monitor.center().x - rect.width() / 2;
                            let cy = monitor.center().y - rect.height() / 2;
                            unsafe {
                                let _ = MoveWindow(hwnd, cx, cy,
                                    rect.width(), rect.height(), true);
                            }
                        }
                    }
                    WindowAction::AlmostMaximize => {
                        let w = (monitor.width() as f64 * 0.9) as i32;
                        let h = (monitor.height() as f64 * 0.9) as i32;
                        let x = monitor.origin.x + (monitor.width() - w) / 2;
                        let y = monitor.origin.y + (monitor.height() - h) / 2;
                        unsafe {
                            let _ = MoveWindow(hwnd, x, y, w, h, true);
                        }
                    }
                    WindowAction::MaximizeHeight => {
                        if let Ok(rect) = self.get_window_rect(window_handle) {
                            unsafe {
                                let _ = MoveWindow(hwnd,
                                    rect.origin.x, monitor.origin.y,
                                    rect.width(), monitor.height(), true);
                            }
                        }
                    }
                    _ => {} // NextDisplay, PreviousDisplayлҠ” лі„лҸ„ мІҳлҰ¬
                }
            }
        }
        Ok(())
    }

    fn get_window_rect(&self, window_handle: u64) -> AppResult<MonitorBounds> {
        let hwnd = self.hwnd_from_u64(window_handle);
        let mut rect = RECT::default();
        unsafe {
            GetWindowRect(hwnd, &mut rect).map_err(|e| ApplicationError::WindowOperation(
                format!("GetWindowRect failed: {}", e)
            ))?;
        }
        Ok(self.rect_to_monitor(rect))
    }
}
```

```rust
#![cfg(windows)]

use windows::Win32::Foundation::{BOOL, LPARAM, POINT, RECT};
use windows::Win32::Graphics::Gdi::{
    EnumDisplayMonitors, GetMonitorInfoW, MonitorFromPoint,
    MonitorEnumProc, MONITOR_DEFAULTTONEAREST, MONITORINFOEXW,
    HDC, HMONITOR,
};

use crate::application::ports::MonitorProvider;
use crate::domain::geometry::MonitorBounds;

pub struct Win32MonitorProvider;

impl Win32MonitorProvider {
    pub fn new() -> Self { Self }
}

impl MonitorProvider for Win32MonitorProvider {
    fn enumerate(&self) -> Vec<MonitorBounds> {
        let monitors: std::sync::Mutex<Vec<MonitorBounds>> = std::sync::Mutex::new(Vec::new());
        let lparam = LPARAM(&monitors as *const _ as isize);

        unsafe {
            let _ = EnumDisplayMonitors(None, None, Some(enum_proc), lparam);
        }

        monitors.into_inner().unwrap_or_default()
    }

    fn monitor_at(&self, x: i32, y: i32) -> MonitorBounds {
        let pt = POINT { x, y };
        unsafe {
            let hmon = MonitorFromPoint(pt, MONITOR_DEFAULTTONEAREST);
            let mut info = MONITORINFOEXW::default();
            info.monitorInfo.cbSize = std::mem::size_of::<MONITORINFOEXW>() as u32;
            if GetMonitorInfoW(hmon, &mut info as *mut _ as *mut _).as_bool() {
                let rc = info.monitorInfo.rcMonitor;
                return MonitorBounds::new(
                    rc.left, rc.top,
                    rc.right - rc.left,
                    rc.bottom - rc.top,
                );
            }
        }
        // нҸҙл°ұ
        MonitorBounds::new(0, 0, 1920, 1080)
    }
}

extern "system" fn enum_proc(
    _hmon: HMONITOR,
    _hdc: HDC,
    rect: *mut RECT,
    lparam: LPARAM,
) -> BOOL {
    let monitors: &std::sync::Mutex<Vec<MonitorBounds>> =
        unsafe { &*(lparam.0 as *const _) };
    let rc = unsafe { *rect };
    let bounds = MonitorBounds::new(
        rc.left, rc.top,
        rc.right - rc.left,
        rc.bottom - rc.top,
    );
    monitors.lock().unwrap().push(bounds);
    BOOL(1) // кі„мҶҚ м—ҙкұ°
}
```

OverlayController кө¬нҳ„мІҙ. Tauriмқҳ AppHandleмқ„ нҶөн•ҙ н”„лЎ нҠём—”л“ң мҳӨлІ„л Ҳмқҙ м°ҪмңјлЎң мқҙлІӨнҠё emit.

```rust
use serde::Serialize;
use std::sync::Mutex;

use crate::application::errors::{ApplicationError, AppResult};
use crate::application::ports::OverlayController;

/// мҳӨлІ„л Ҳмқҙ мғҒнғң вҖ” н”„лЎ нҠём—”л“ңм—җ emitлҗҳлҠ” нҺҳмқҙлЎңл“ң
#[derive(Debug, Clone, Serialize)]
pub struct OverlayState {
    pub visible: bool,
    pub center: Option<(i32, i32)>,
    pub sector_count: u8,
    pub cursor_pos: Option<(i32, i32)>,
    pub active_sector: Option<u8>,
    pub snap_preview: Option<(i32, i32, i32, i32)>, // x, y, w, h
}

impl Default for OverlayState {
    fn default() -> Self {
        Self {
            visible: false,
            center: None,
            sector_count: 8,
            cursor_pos: None,
            active_sector: None,
            snap_preview: None,
        }
    }
}

/// Tauri AppHandleмқ„ мӮ¬мҡ©н•ҳлҠ” OverlayController.
/// мғҒнғңлҘј к°ұмӢ н•ҳкі  н”„лЎ нҠём—”л“ңм—җ мқҙлІӨнҠёлЎң emit.
pub struct TauriOverlay {
    state: Mutex<OverlayState>,
    /// Tauri AppHandle (лӮҳмӨ‘м—җ мЈјмһ…). нҳ„мһ¬лҠ” мғҒнғңл§Ң кҙҖлҰ¬.
    /// мӢӨм ң emitмқҖ presentation кі„мёөм—җм„ң Tauri AppHandleлЎң мҲҳн–ү.
    emit_fn: Mutex<Option<Box<dyn Fn(&OverlayState) + Send + Sync>>>,
}

impl TauriOverlay {
    pub fn new() -> Self {
        Self {
            state: Mutex::new(OverlayState::default()),
            emit_fn: Mutex::new(None),
        }
    }

    /// emit н•ЁмҲҳ мЈјмһ… (presentation кі„мёөм—җм„ң AppHandleлЎң emitн•ҳлҸ„лЎқ)
    pub fn set_emitter(&self, f: impl Fn(&OverlayState) + Send + Sync + 'static) {
        *self.emit_fn.lock().unwrap() = Some(Box::new(f));
    }

    fn update_and_emit(&self, f: impl FnOnce(&mut OverlayState)) -> AppResult<()> {
        {
            let mut state = self.state.lock().unwrap();
            f(&mut state);
        }
        let state = self.state.lock().unwrap();
        if let Some(emit) = self.emit_fn.lock().unwrap().as_ref() {
            emit(&state);
        }
        Ok(())
    }
}

impl OverlayController for TauriOverlay {
    fn show_reticle(&self, center_x: i32, center_y: i32, sector_count: u8) -> AppResult<()> {
        self.update_and_emit(|s| {
            s.visible = true;
            s.center = Some((center_x, center_y));
            s.sector_count = sector_count;
            s.active_sector = None;
            s.snap_preview = None;
        })
    }

    fn update_cursor_indicator(&self, x: i32, y: i32) -> AppResult<()> {
        self.update_and_emit(|s| {
            s.cursor_pos = Some((x, y));
        })
    }

    fn highlight_sector(&self, sector: u8) -> AppResult<()> {
        self.update_and_emit(|s| {
            s.active_sector = Some(sector);
        })
    }

    fn show_snap_preview(&self, x: i32, y: i32, width: i32, height: i32) -> AppResult<()> {
        self.update_and_emit(|s| {
            s.snap_preview = Some((x, y, width, height));
        })
    }

    fn hide(&self) -> AppResult<()> {
        self.update_and_emit(|s| {
            s.visible = false;
            s.center = None;
            s.cursor_pos = None;
            s.active_sector = None;
            s.snap_preview = None;
        })
    }
}
```

```rust
use crate::domain::model::{Direction, ModifierMode};

/// мһ…л Ҙ мқҙлІӨнҠё мў…лҘҳ (нӣ… мҪңл°ұм—җм„ң ліҖнҷҳ)
#[derive(Debug, Clone, PartialEq)]
pub enum InputEvent {
    /// modifier мЎ°н•©мқҙ лҲҢлҰј (throw нҠёлҰ¬кұ°)
    ModifierPressed { modifiers: Vec<String> },
    /// modifier мЎ°н•©мқҙ л–јм–ҙм§җ
    ModifierReleased { modifiers: Vec<String>, cancel: bool },
    /// л§Ҳмҡ°мҠӨ мқҙлҸҷ (delta н”Ҫм…Җ)
    MouseMoved { delta_x: f64, delta_y: f64 },
    /// л°©н–ҘнӮӨ
    ArrowKey { direction: Direction, modifiers: Vec<String> },
}

/// лқјмҡ°нҢ… лҢҖмғҒ
#[derive(Debug, Clone, PartialEq)]
pub enum RouteTarget {
    /// Window Throw (SnapService)
    SnapService,
    /// нӮӨліҙл“ң мҠӨлғ… (KeyboardService)
    KeyboardService(Direction),
    /// л¬ҙмӢң (мқҙлІӨнҠё м „лӢ¬)
    Ignore,
    /// Win+л°©н–ҘнӮӨ swallow (OS кё°ліё лҸҷмһ‘ м°ЁлӢЁ)
    SwallowOsSnap(Direction),
}

/// modifier л¬ёмһҗм—ҙ л№„көҗ н—¬нҚј
fn modifiers_match(pressed: &[String], expected: &[String]) -> bool {
    if pressed.len() != expected.len() {
        return false;
    }
    let mut pressed_sorted = pressed.to_vec();
    let mut expected_sorted = expected.to_vec();
    pressed_sorted.sort();
    expected_sorted.sort();
    pressed_sorted == expected_sorted
}

/// мһ…л Ҙ мқҙлІӨнҠёлҘј лқјмҡ°нҢ… лҢҖмғҒмңјлЎң л¶„лҘҳ.
///
/// throw_modifiers: Throw(м»Өм„ң л¬ҙлёҢ)мҡ© modifier мЎ°н•©
/// kb_modifiers: нӮӨліҙл“ң мҠӨлғ…мҡ© modifier мЎ°н•©
/// kb_mode: нӮӨліҙл“ң мҠӨлғ… modifier лӘЁл“ң
pub fn classify(
    event: &InputEvent,
    throw_modifiers: &[String],
    kb_modifiers: &[String],
    kb_mode: ModifierMode,
) -> RouteTarget {
    match event {
        InputEvent::ModifierPressed { modifiers } => {
            // throw modifierк°Җ лҲҢл ёмңјл©ҙ SnapServiceлЎң
            if modifiers_match(modifiers, throw_modifiers) {
                RouteTarget::SnapService
            } else {
                RouteTarget::Ignore
            }
        }
        InputEvent::MouseMoved { .. } => {
            // SnapServiceк°Җ нҷңм„ұ мғҒнғңмқј л•Ңл§Ң мқҳлҜё мһҲмқҢ
            // мӢӨм ң нҷңм„ұ м—¬л¶ҖлҠ” SnapService лӮҙл¶Җ мғҒнғңк°Җ нҢҗлӢЁ
            RouteTarget::SnapService
        }
        InputEvent::ModifierReleased { modifiers, .. } => {
            if modifiers_match(modifiers, throw_modifiers) {
                RouteTarget::SnapService
            } else {
                RouteTarget::Ignore
            }
        }
        InputEvent::ArrowKey { direction, modifiers } => {
            match kb_mode {
                ModifierMode::OverrideOs => {
                    // Win нӮӨл§Ң мһҲкі  л°©н–ҘнӮӨл©ҙ swallow + KeyboardService
                    if modifiers.iter().any(|m| m == "Win") {
                        RouteTarget::SwallowOsSnap(*direction)
                    } else {
                        RouteTarget::Ignore
                    }
                }
                ModifierMode::Shared => {
                    // throw modifierмҷҖ к°ҷмқҖ мЎ°н•©
                    if modifiers_match(modifiers, throw_modifiers) {
                        RouteTarget::KeyboardService(*direction)
                    } else {
                        RouteTarget::Ignore
                    }
                }
                ModifierMode::Separate => {
                    if modifiers_match(modifiers, kb_modifiers) {
                        RouteTarget::KeyboardService(*direction)
                    } else {
                        RouteTarget::Ignore
                    }
                }
            }
        }
    }
}
```
